// Define a function to calculate the factorial of a number
function factorial(num) {
  // If the number is less than or equal to 1, return 1
  if (num <= 1) {
    return 1;
  }
  // Otherwise, return the number multiplied by the factorial of the number - 1
  return num * factorial(num - 1);
}

// Define a variable to store the number
let num = 5;

// Calculate the factorial of the number and store the result in a variable
let result = factorial(num);

// Log the result to the console
console.log(result); // Output: 120
